130  // Arnoldi factorization starting from step-k
131  void factorize_from(int from_k, int to_m, const Vector &fk)
132  {
133  if(to_m <= from_k) return;
134 
135  fac_f = fk;
136 
137  Vector w(dim_n);
138  Scalar beta = 0.0;
139  // Keep the upperleft k x k submatrix of H and set other elements to 0
140  fac_H.rightCols(ncv - from_k).setZero();
141  fac_H.block(from_k, 0, ncv - from_k, from_k).setZero();
142  for(int i = from_k; i <= to_m - 1; i++)
143  {
144  beta = fac_f.norm();
145  fac_V.col(i).noalias() = fac_f / beta; // The (i+1)-th column
146  fac_H(i, i - 1) = beta;
147 
148  // w = A * v, v = fac_V.col(i)
149  op->perform_op(&fac_V(0, i), w.data());
150  nmatop++;
151 
152  // First i+1 columns of V
153  MapMat Vs(fac_V.data(), dim_n, i + 1);
154  // h = fac_H(0:i, i)
155  MapVec h(&fac_H(0, i), i + 1);
156  h.noalias() = Vs.transpose() * w;
157 
158  fac_f = w - Vs * h;
159  // Correct f if it is not orthogonal to V
160  // Typically the largest absolute value occurs in
161  // the first element, i.e., <v1, f>, so we use this
162  // to test the orthogonality
163  Scalar v1f = fac_f.dot(fac_V.col(0));
164  if(v1f > prec || v1f < -prec)
165  {
166  Vector Vf(i + 1);
167  Vf.tail(i) = fac_V.block(0, 1, dim_n, i).transpose() * fac_f;
168  Vf[0] = v1f;
169  fac_f -= Vs * Vf;
170  }
171  }
172  }

184  // Implicitly restarted Arnoldi factorization
185  void restart(int k)
186  {
187  if(k >= ncv)
188  return;
189 
190  DoubleShiftQR<Scalar> decomp_ds(ncv);
191  UpperHessenbergQR<Scalar> decomp_hb;
192  Matrix Q = Matrix::Identity(ncv, ncv);
193 
194  for(int i = k; i < ncv; i++)
195  {
196  if(is_complex(ritz_val[i], prec) && is_conj(ritz_val[i], ritz_val[i + 1], prec))
197  {
198  // H - mu * I = Q1 * R1
199  // H <- R1 * Q1 + mu * I = Q1' * H * Q1
200  // H - conj(mu) * I = Q2 * R2
201  // H <- R2 * Q2 + conj(mu) * I = Q2' * H * Q2
202  //
203  // (H - mu * I) * (H - conj(mu) * I) = Q1 * Q2 * R2 * R1 = Q * R
204  Scalar s = 2 * ritz_val[i].real();
205  Scalar t = std::norm(ritz_val[i]);
206 
207  decomp_ds.compute(fac_H, s, t);
208 
209  // Q -> Q * Qi
210  decomp_ds.apply_YQ(Q);
211  // H -> Q'HQ
212  // Matrix Q = Matrix::Identity(ncv, ncv);
213  // decomp_ds.apply_YQ(Q);
214  // fac_H = Q.transpose() * fac_H * Q;
215  fac_H = decomp_ds.matrix_QtHQ();
216 
217  i++;
218  } else {
219  // QR decomposition of H - mu * I, mu is real
220  fac_H.diagonal().array() -= ritz_val[i].real();
221  decomp_hb.compute(fac_H);
222 
223  // Q -> Q * Qi
224  decomp_hb.apply_YQ(Q);
225  // H -> Q'HQ = RQ + mu * I
226  fac_H = decomp_hb.matrix_RQ();
227  fac_H.diagonal().array() += ritz_val[i].real();
228  }
229  }
230  // V -> VQ, only need to update the first k+1 columns
231  // Q has some elements being zero
232  // The first (ncv - k + i) elements of the i-th column of Q are non-zero
233  Matrix Vs(dim_n, k + 1);
234  int nnz;
235  for(int i = 0; i < k; i++)
236  {
237  nnz = ncv - k + i + 1;
238  MapMat V(fac_V.data(), dim_n, nnz);
239  MapVec q(&Q(0, i), nnz);
240  Vs.col(i).noalias() = V * q;
241  }
242  Vs.col(k).noalias() = fac_V * Q.col(k);
243  fac_V.leftCols(k + 1).noalias() = Vs;
244 
245  Vector fk = fac_f * Q(ncv - 1, k - 1) + fac_V.col(k) * fac_H(k, k - 1);
246  factorize_from(k, ncv, fk);
247  retrieve_ritzpair();
248  }

250  // Calculate the number of converged Ritz values
251  int num_converged(Scalar tol)
252  {
253  // thresh = tol * max(prec, abs(theta)), theta for ritz value
254  Array thresh = tol * ritz_val.head(nev).array().abs().max(prec);
255  Array resid = ritz_vec.template bottomRows<1>().transpose().array().abs() * fac_f.norm();
256  // Converged "wanted" ritz values
257  ritz_conv = (resid < thresh);
258 
259  return ritz_conv.cast<int>().sum();
260  }

262  // Return the adjusted nev for restarting
263  int nev_adjusted(int nconv)
264  {
265  int nev_new = nev;
266 
267  // Increase nev by one if ritz_val[nev - 1] and
268  // ritz_val[nev] are conjugate pairs
269  if(is_complex(ritz_val[nev - 1], prec) &&
270  is_conj(ritz_val[nev - 1], ritz_val[nev], prec))
271  {
272  nev_new = nev + 1;
273  }
274  // Adjust nev_new again, according to dnaup2.f line 660~674 in ARPACK
275  nev_new = nev_new + std::min(nconv, (ncv - nev_new) / 2);
276  if(nev_new == 1 && ncv >= 6)
277  nev_new = ncv / 2;
278  else if(nev_new == 1 && ncv > 3)
279  nev_new = 2;
280 
281  if(nev_new > ncv - 2)
282  nev_new = ncv - 2;
283 
284  // Examine conjugate pairs again
285  if(is_complex(ritz_val[nev_new - 1], prec) &&
286  is_conj(ritz_val[nev_new - 1], ritz_val[nev_new], prec))
287  {
288  nev_new++;
289  }
290 
291  return nev_new;
292  }

294  // Retrieve and sort ritz values and ritz vectors
295  void retrieve_ritzpair()
296  {
297  UpperHessenbergEigen<Scalar> decomp(fac_H);
298  ComplexVector evals = decomp.eigenvalues();
299  ComplexMatrix evecs = decomp.eigenvectors();
300 
301  SortEigenvalue<Complex, SelectionRule> sorting(evals.data(), evals.size());
302  std::vector<int> ind = sorting.index();
303 
304  // Copy the ritz values and vectors to ritz_val and ritz_vec, respectively
305  for(int i = 0; i < ncv; i++)
306  {
307  ritz_val[i] = evals[ind[i]];
308  }
309  for(int i = 0; i < nev; i++)
310  {
311  ritz_vec.col(i) = evecs.col(ind[i]);
312  }
313  }

315 protected:
316  // Sort the first nev Ritz pairs in the specified order
317  // This is used to return the final results
318  virtual void sort_ritzpair(int sort_rule)
319  {
320  // First make sure that we have a valid index vector
321  SortEigenvalue<Complex, LARGEST_MAGN> sorting(ritz_val.data(), nev);
322  std::vector<int> ind = sorting.index();
323 
324  switch(sort_rule)
325  {
326  case LARGEST_MAGN:
327  break;
328  case LARGEST_REAL:
329  {
330  SortEigenvalue<Complex, LARGEST_REAL> sorting(ritz_val.data(), nev);
331  ind = sorting.index();
332  }
333  break;
334  case LARGEST_IMAG:
335  {
336  SortEigenvalue<Complex, LARGEST_IMAG> sorting(ritz_val.data(), nev);
337  ind = sorting.index();
338  }
339  break;
340  case SMALLEST_MAGN:
341  {
342  SortEigenvalue<Complex, SMALLEST_MAGN> sorting(ritz_val.data(), nev);
343  ind = sorting.index();
344  }
345  break;
346  case SMALLEST_REAL:
347  {
348  SortEigenvalue<Complex, SMALLEST_REAL> sorting(ritz_val.data(), nev);
349  ind = sorting.index();
350  }
351  break;
352  case SMALLEST_IMAG:
353  {
354  SortEigenvalue<Complex, SMALLEST_IMAG> sorting(ritz_val.data(), nev);
355  ind = sorting.index();
356  }
357  break;
358  default:
359  throw std::invalid_argument("unsupported sorting rule");
360  }
361 
362  ComplexVector new_ritz_val(ncv);
363  ComplexMatrix new_ritz_vec(ncv, nev);
364  BoolArray new_ritz_conv(nev);
365 
366  for(int i = 0; i < nev; i++)
367  {
368  new_ritz_val[i] = ritz_val[ind[i]];
369  new_ritz_vec.col(i) = ritz_vec.col(ind[i]);
370  new_ritz_conv[i] = ritz_conv[ind[i]];
371  }
372 
373  ritz_val.swap(new_ritz_val);
374  ritz_vec.swap(new_ritz_vec);
375  ritz_conv.swap(new_ritz_conv);
376  }

422  {
423  // Reset all matrices/vectors to zero
424  fac_V.resize(dim_n, ncv);
425  fac_H.resize(ncv, ncv);

488  int compute(int maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_MAGN)
489  {
490  // The m-step Arnoldi factorization
491  factorize_from(1, ncv, fac_f);
492  retrieve_ritzpair();
493  // Restarting
494  int i, nconv = 0, nev_adj;
495  for(i = 0; i < maxit; i++)
496  {
497  nconv = num_converged(tol);
498  if(nconv >= nev)
499  break;
500 
501  nev_adj = nev_adjusted(nconv);
502  restart(nev_adj);
503  }
504  // Sorting results
505  sort_ritzpair(sort_rule);
506 
507  niter += i + 1;
508 
509  return std::min(nev, nconv);
510  }

686 template <typename Scalar = double,
687  int SelectionRule = LARGEST_MAGN,
688  typename OpType = DenseGenComplexShiftSolve<double> >
689 class GenEigsComplexShiftSolver: public GenEigsSolver<Scalar, SelectionRule, OpType>
690 {
691 private:
692  typedef Eigen::Array<Scalar, Eigen::Dynamic, 1> Array;
693  typedef std::complex<Scalar> Complex;
694  typedef Eigen::Array<Complex, Eigen::Dynamic, 1> ComplexArray;
695 
696  Scalar sigmar;
697  Scalar sigmai;
698 
699  // First transform back the ritz values, and then sort
700  void sort_ritzpair(int sort_rule)
701  {
702  // The eigenvalus we get from the iteration is
703  // nu = 0.5 * (1 / (lambda - sigma)) + 1 / (lambda - conj(sigma)))
704  // So the eigenvalues of the original problem is
705  // 1 \pm sqrt(1 - 4 * nu^2 * sigmai^2)
706  // lambda = sigmar + -----------------------------------
707  // 2 * nu
708  // We need to pick up the correct root
709  // Let vi be the i-th eigenvector, then A * vi = lambdai * vi
710  // and inv(A - r * I) * vi = 1 / (lambdai - r) * vi
711  // where r is any real value.
712  // We can use this identity to back-solve lambdai
713 
714  // Select an arbitrary real shift value
715  Scalar r = sigmar + std::sin(sigmar);
716  this->op->set_shift(r, 0);
717 
718  // Calculate inv(A - r * I) * vi
719  ComplexArray v;
720  Array v_real, v_imag;
721  Array lhs_real(this->dim_n), lhs_imag(this->dim_n);
722  Scalar eps = std::pow(std::numeric_limits<Scalar>::epsilon(), Scalar(2.0) / 3);
723  for(int i = 0; i < this->nev; i++)
724  {
725  v = this->fac_V * this->ritz_vec.col(i);
726  v_real = v.real();
727  v_imag = v.imag();
728 
729  this->op->perform_op(v_real.data(), lhs_real.data());
730  this->op->perform_op(v_imag.data(), lhs_imag.data());
731 
732  Complex lambdai = Complex(v_real[0], v_imag[0]) / Complex(lhs_real[0], lhs_imag[0]) +
733  Complex(r, 0);
734  this->ritz_val[i] = lambdai;
735 
736  if(std::abs(lambdai.imag()) > eps)
737  {
738  this->ritz_val[i + 1] = std::conj(lambdai);
739  i++;
740  }
741  }
742 
743  GenEigsSolver<Scalar, SelectionRule, OpType>::sort_ritzpair(sort_rule);
744  }
745 public:
765  GenEigsComplexShiftSolver(OpType *op_, int nev_, int ncv_, Scalar sigmar_, Scalar sigmai_) :
766  GenEigsSolver<Scalar, SelectionRule, OpType>(op_, nev_, ncv_),
767  sigmar(sigmar_), sigmai(sigmai_)
