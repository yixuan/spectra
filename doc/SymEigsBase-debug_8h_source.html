97  // Implicitly restarted Lanczos factorization
98  void restart(Index k, SortRule selection)
99  {
100  using std::abs;
101 
102  if (k >= m_ncv)
103  return;
104 
105  TridiagQR<Scalar> decomp(m_ncv);
106  Matrix Q = Matrix::Identity(m_ncv, m_ncv);
107 
108  // Apply large shifts first
109  const int nshift = m_ncv - k;
110  Vector shifts = m_ritz_val.tail(nshift);
111  std::sort(shifts.data(), shifts.data() + nshift, [](const Scalar& v1, const Scalar& v2) { return abs(v1) > abs(v2); });
112 
113  /*constexpr Scalar m_eps = TypeTraits<Scalar>::epsilon();
114  const Scalar m_eps23 = pow(m_eps, Scalar(2) / 3);
115  Vector shifts = m_ritz_val.segment(k, m_ncv - k);
116  Vector bound = m_ritz_est.segment(k, m_ncv - k);
117  using Pair = std::pair<Scalar, Scalar>;
118  std::vector<Pair> pairs(m_ncv - k);
119  for (int i = 0; i < m_ncv - k; i++)
120  {
121  const Scalar denom = std::max(m_eps23, abs(shifts[i]));
122  pairs.emplace_back(shifts[i], std::abs(bound[i]) / denom);
123  }
124  std::sort(pairs.begin(), pairs.end(), [](const Pair& p1, const Pair& p2) { return p1.second > p2.second; });*/
125 
126  for (Index i = 0; i < nshift; i++)
127  {
128  // QR decomposition of H-mu*I, mu is the shift
129  decomp.compute(m_fac.matrix_H(), shifts[i]);
130 
131  // Q -> Q * Qi
132  decomp.apply_YQ(Q);
133  // H -> Q'HQ
134  // Since QR = H - mu * I, we have H = QR + mu * I
135  // and therefore Q'HQ = RQ + mu * I
136  m_fac.compress_H(decomp);
137  }
138 
139  m_fac.compress_V(Q);
140  m_fac.factorize_from(k, m_ncv, m_nmatop);
141 
142  /*if (ct % 100 == 0 || ct >= 9990)
143  {
144  std::cout << "i = " << ct << std::endl;
145  m_fac.debug();
146  }*/
147 
148  retrieve_ritzpair(selection);
149  }

151  // Calculates the number of converged Ritz values
152  Index num_converged(const Scalar& tol)
153  {
154  using std::pow;
155 
156  // The machine precision, ~= 1e-16 for the "double" type
157  constexpr Scalar eps = TypeTraits<Scalar>::epsilon();
158  // std::pow() is not constexpr, so we do not declare eps23 to be constexpr
159  // But most compilers should be able to compute eps23 at compile time
160  const Scalar eps23 = pow(eps, Scalar(2) / 3);
161 
162  // thresh = tol * max(eps23, abs(theta)), theta for Ritz value
163  Array thresh = tol * m_ritz_val.head(m_nev).array().abs().max(eps23);
164  Array resid = m_ritz_est.head(m_nev).array().abs() * m_fac.f_norm();
165  // Converged "wanted" Ritz values
166  m_ritz_conv = (resid < thresh);
167 
168  /*if (ct % 100 == 0 || ct >= 9990)
169  {
170  std::cout << "i = " << ct << std::endl;
171  std::cout << "ritzval_diff = " << (m_ritz_val.array() - 0.49348022005445374).matrix().transpose() << std::endl;
172  std::cout << "thresh = " << thresh.matrix().transpose() << std::endl;
173  std::cout << "resid = " << resid.matrix().transpose() << std::endl
174  << std::endl;
175  }*/
176 
177  return m_ritz_conv.count();

180  // Returns the adjusted nev for restarting
181  Index nev_adjusted(Index nconv)
182  {
183  using std::abs;
184 
185  // A very small value, but 1.0 / near_0 does not overflow
186  // ~= 1e-307 for the "double" type
187  constexpr Scalar near_0 = TypeTraits<Scalar>::min() * Scalar(10);
188 
189  Index nev_new = m_nev;
190  for (Index i = m_nev; i < m_ncv; i++)
191  if (abs(m_ritz_est[i]) < near_0)
192  nev_new++;
193 
194  // Adjust nev_new, according to dsaup2.f line 677~684 in ARPACK
195  nev_new += std::min(nconv, (m_ncv - nev_new) / 2);
196  if (nev_new == 1 && m_ncv >= 6)
197  nev_new = m_ncv / 2;
198  else if (nev_new == 1 && m_ncv > 2)
199  nev_new = 2;
200 
201  if (nev_new > m_ncv - 1)
202  nev_new = m_ncv - 1;
203 
204  return nev_new;
205  }

207  // Retrieves and sorts Ritz values and Ritz vectors
208  void retrieve_ritzpair(SortRule selection)
209  {
210  TridiagEigen<Scalar> decomp(m_fac.matrix_H());
211  const Vector& evals = decomp.eigenvalues();
212  const Matrix& evecs = decomp.eigenvectors();
213 
214  // Sort Ritz values and put the wanted ones at the beginning
215  std::vector<Index> ind;
216  switch (selection)
217  {
218  case SortRule::LargestMagn:
219  {
220  SortEigenvalue<Scalar, SortRule::LargestMagn> sorting(evals.data(), m_ncv);
221  sorting.swap(ind);
222  break;
223  }
224  case SortRule::BothEnds:
225  case SortRule::LargestAlge:
226  {
227  SortEigenvalue<Scalar, SortRule::LargestAlge> sorting(evals.data(), m_ncv);
228  sorting.swap(ind);
229  break;
230  }
231  case SortRule::SmallestMagn:
232  {
233  SortEigenvalue<Scalar, SortRule::SmallestMagn> sorting(evals.data(), m_ncv);
234  sorting.swap(ind);
235  break;
236  }
237  case SortRule::SmallestAlge:
238  {
239  SortEigenvalue<Scalar, SortRule::SmallestAlge> sorting(evals.data(), m_ncv);
240  sorting.swap(ind);
241  break;
242  }
243  default:
244  throw std::invalid_argument("unsupported selection rule");
245  }
246 
247  // For SortRule::BothEnds, the eigenvalues are sorted according to the
248  // SortRule::LargestAlge rule, so we need to move those smallest values to the left
249  // The order would be
250  // Largest => Smallest => 2nd largest => 2nd smallest => ...
251  // We keep this order since the first k values will always be
252  // the wanted collection, no matter k is nev_updated (used in restart())
253  // or is nev (used in sort_ritzpair())
254  if (selection == SortRule::BothEnds)
255  {
256  std::vector<Index> ind_copy(ind);
257  for (Index i = 0; i < m_ncv; i++)
258  {
259  // If i is even, pick values from the left (large values)
260  // If i is odd, pick values from the right (small values)
261  if (i % 2 == 0)
262  ind[i] = ind_copy[i / 2];
263  else
264  ind[i] = ind_copy[m_ncv - 1 - i / 2];
265  }
266  }
267 
268  // Copy the Ritz values and vectors to m_ritz_val and m_ritz_vec, respectively
269  for (Index i = 0; i < m_ncv; i++)
270  {
271  m_ritz_val[i] = evals[ind[i]];
272  m_ritz_est[i] = evecs(m_ncv - 1, ind[i]);
273  }
274  for (Index i = 0; i < m_nev; i++)
275  {
276  m_ritz_vec.col(i).noalias() = evecs.col(ind[i]);
277  }
278  }

281  // Sorts the first nev Ritz pairs in the specified order
282  // This is used to return the final results
283  virtual void sort_ritzpair(SortRule sort_rule)
284  {
285  std::vector<Index> ind;
286  switch (sort_rule)
287  {
288  case SortRule::LargestAlge:
289  {
290  SortEigenvalue<Scalar, SortRule::LargestAlge> sorting(m_ritz_val.data(), m_nev);
291  sorting.swap(ind);
292  break;
293  }
294  case SortRule::LargestMagn:
295  {
296  SortEigenvalue<Scalar, SortRule::LargestMagn> sorting(m_ritz_val.data(), m_nev);
297  sorting.swap(ind);
298  break;
299  }
300  case SortRule::SmallestAlge:
301  {
302  SortEigenvalue<Scalar, SortRule::SmallestAlge> sorting(m_ritz_val.data(), m_nev);
303  sorting.swap(ind);
304  break;
305  }
306  case SortRule::SmallestMagn:
307  {
308  SortEigenvalue<Scalar, SortRule::SmallestMagn> sorting(m_ritz_val.data(), m_nev);
309  sorting.swap(ind);
310  break;
311  }
312  default:
313  throw std::invalid_argument("unsupported sorting rule");
314  }
315 
316  Vector new_ritz_val(m_ncv);
317  Matrix new_ritz_vec(m_ncv, m_nev);
318  BoolArray new_ritz_conv(m_nev);
319 
320  for (Index i = 0; i < m_nev; i++)
321  {
322  new_ritz_val[i] = m_ritz_val[ind[i]];
323  new_ritz_vec.col(i).noalias() = m_ritz_vec.col(ind[i]);
324  new_ritz_conv[i] = m_ritz_conv[ind[i]];
325  }
326 
327  m_ritz_val.swap(new_ritz_val);
328  m_ritz_vec.swap(new_ritz_vec);
329  m_ritz_conv.swap(new_ritz_conv);
330  }

445  Index compute(SortRule selection = SortRule::LargestMagn, Index maxit = 1000,
446  Scalar tol = 1e-10, SortRule sorting = SortRule::LargestAlge)
447  {
448  // The m-step Lanczos factorization
449  m_fac.factorize_from(1, m_ncv, m_nmatop);
450  retrieve_ritzpair(selection);
451  // Restarting
452  Index i, nconv = 0, nev_adj;
453  for (i = 0; i < maxit; i++)
454  {
455  ct = i;
456  nconv = num_converged(tol);
457  if (nconv >= m_nev)
458  break;
459 
460  nev_adj = nev_adjusted(nconv);
461  restart(nev_adj, selection);
462  }
463  // Sorting results
464  sort_ritzpair(sorting);
465 
466  m_niter += i + 1;
467  m_info = (nconv >= m_nev) ? CompInfo::Successful : CompInfo::NotConverging;
468 
469  return std::min(m_nev, nconv);
470  }
